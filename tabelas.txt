CREATE TABLE `financeiro`.`new_table` (
  `id` INT NOT NULL AUTO_INCREMENT COMMENT '',
  `nome` VARCHAR(45) NULL COMMENT '',
  `apelido` VARCHAR(45) NULL COMMENT '',
  `data_admissao` DATE NULL COMMENT '',
  `tipo` INT NULL COMMENT '',
  PRIMARY KEY (`id`)  COMMENT '');





CREATE TABLE `financeiro`.`ligacao` (
  `id` INT NOT NULL AUTO_INCREMENT COMMENT '',
  `telefone` VARCHAR(45) NULL COMMENT '',
  `grau_interesse` INT NULL COMMENT '',
  `data` DATETIME NULL COMMENT '',
  `situacao` INT NULL COMMENT '',
  `observacao` VARCHAR(500) NULL COMMENT '',
  `id_acessor` INT NULL COMMENT '',
  PRIMARY KEY (`id`)  COMMENT '',
  INDEX `id_ligacao_acessor_idx` (`id_acessor` ASC)  COMMENT '',
  CONSTRAINT `id_ligacao_acessor`
    FOREIGN KEY (`id_acessor`)
    REFERENCES `financeiro`.`acessor` (`id`)
    ON DELETE NO ACTION
    ON UPDATE NO ACTION);
