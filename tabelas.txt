CREATE TABLE `financeiro`.`ligacao` (
  `id` INT NOT NULL AUTO_INCREMENT COMMENT '',
  `telefone` VARCHAR(45) NULL COMMENT '',
  `grau_interesse` INT NULL COMMENT '',
  `data` DATETIME NULL COMMENT '',
  `situacao` INT NULL COMMENT '',
  `observacao` VARCHAR(500) NULL COMMENT '',

  INDEX `id_ligacao_acessor_idx` (`id_acessor` ASC)  COMMENT '',
